Web Component里面Custom Element的入门指南：

通过document.registerElement('x-component');注册之后就能<x-component></x-component>创建自定义元素，或者通过new XComponent();来创建。还可以通过传输proto来为自定义元素增加特性：

var proto = Object.create(HTMLElement.prototype);
proto.alert = function() {
  alert('我是' + this.name);
};
document.registerElement('x-component', {
  prototype: proto
});
